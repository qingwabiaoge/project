# 小程序data method,生命周期钩子函数


```
App({
  globalData: 1,

  attached: function () { }
  moved: function () { }
  detached: function () {}
  onTap: function () {

    }
})
```
```
Page({
 data: {
    msg:""
  },
// 生命周期函数，可以为函数，或一个在methods段中定义的方法名
  attached: function () { }
  moved: function () { }
  detached: function () {}
  onTap: function () {

    }
})

```
```
Component({
 data: {
    msg:""
  },
// 生命周期函数，可以为函数，或一个在methods段中定义的方法名
  attached: function () { }
  moved: function () { }
  detached: function () {}
  methods: {
    onTap: function () {
     
    }
  }
})
```
