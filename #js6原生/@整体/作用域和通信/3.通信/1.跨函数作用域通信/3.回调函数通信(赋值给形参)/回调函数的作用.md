# 回调函数的作用 


    function f(i,fn) {
        setTimeout(() => {
            console.log("等待我执行完后你再执行")
            fn(i)//钩子函数
        }, 2000)
    }

    f('回调函数开始执行',(i)=>console.log(i))
    console.log('主程序没有堵塞')
