
# 定义

在函数外部能修改这个函数的内部变量 叫闭包

```javascript
    function outer(){
        var i=1
        console.log('outer（）中的i:'+i)//只运行了一次,outer()调用
        return function (){
            i++ // 因为使用栈内存变量i , i的变量对象无法出栈, 每运行一次i加1
            console.log('闭包里的i:'+i)
        };

    }
    var outObj = outer();//运行一次
    outObj();//&2，执行i += 1后，val还在
    outObj();//&3
    outObj = null;//val 被回收
    var outObj1 = outer();
    outObj1();//&2
    outObj1();//*3
```
