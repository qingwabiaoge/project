### vue组件的$event可以自定义任何值 

```
    <cl @click="fn($event)"></cl>
----------------------------------
    var cl = {
        template:`<div @click="click">事件</div>`,
        methods: {
           click:function () {

               this.$emit('click', 123)
           }
        }

    }
    const vm = new Vue({
        el: '#app',
        components: {cl},
        data: {
            msg: 'msg'
        },
        methods:{

            fn($event){
                console.log($event)
            }

        }
    })
```
