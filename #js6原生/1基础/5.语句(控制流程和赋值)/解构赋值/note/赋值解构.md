# 解构赋值

实质是一种对应相等关系

### Object解构

```js
let {a:a,b:b,c:c}
    {a,b,c}
    {a,b}
    {a,b,c,d}
    {a,b,c,d=4}
    {a,...obj}
    {...obj}={a:1,b:2,c:3}
```

```js
let i

//i===undefine
obj={}

//obj.a===undefine

const {a,b}={a:1}

//b===undefine
```

### Array解构

````js
let
[a,b,c]
[a, ,c]
[a,b]
[a,b,c,d]
[a,b,c,d=4]
[a,...arr]
[...arr]=[1,2,3]
        ='123'
````

```js
const [a,b]=[1]
//b===undifine
const arr=[]
//arr[0]===undifine
```

### 类数组和数组同等待遇

```js
let

[a,b,c,d]

```

```js
let {m,n} = 'ab'//封装成对象类型{'0':'a','1':'b'}
 console.log(m,n) //undefined undefined
```
### 函数参数解构

````js
function f({a,b}) {
    console.log(a,b)//1,2
}

f({a:1,b:2})

//解释
//1形参复制给实参 let {a,b}={a:1,b:2}
//2 函数体里得到了 a,b
````
### import export
//import.js
````js
 var a = 'Michael';
 var b = 'Jackson';
 var c = 1958;
 export{a,b,c} //等价于{a: 'Michael', b: 'Jackson', c: 1958}
````
//export.js
```js
import {a, b, c} from './2.js';
// 实质const {a, b, c}={a: 'Michael', b: 'Jackson', c: 1958}
```
