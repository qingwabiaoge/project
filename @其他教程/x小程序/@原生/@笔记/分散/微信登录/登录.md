## 1. 放到app.js onload 加载

```
  wx.login({
      success: function (res) {
      
        if (res.code) {
         console.log(res)
          //发起网络请求
          wx.request({
            url: config.loginUrl,
            data: {
              code: res.code
            }
          })
        } else {
          console.log('登录失败！' + res.errMsg)
        }
      }
    });
```
