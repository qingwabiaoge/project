# js对象的属性值的变量类型
```js
Vue.component('my-component', {
  props: {
//1基本类型当属性值----------------------------------------
    // 基础的类型检查 (`null` 和 `undefined` 会通过任何类型验证)
    propA: Number,
    // 多个可能的类型
      
//2数组当属性值---------------------------------------------------    
    propB: [String, Number],
    // 必填的字符串
      
//3对象当属性值---------------------------------------------------    
    propC: {
      type: String,
      required: true
    },
    // 带有默认值的数字
    propD: {
      type: Number,
      default: 100
    },
      
             
    // 带有默认值的对象
    propE: {
      type: Object,
      // 对象或数组默认值必须从一个工厂函数获取
//4函数当属性值---------------------------------------------------           
      default: function () {
        return { message: 'hello' }
      }
    },
    // 自定义验证函数
    propF: {
      validator: function (value) {
        // 这个值必须匹配下列字符串中的一个
        return ['success', 'warning', 'danger'].indexOf(value) !== -1
      }
    }
  }
})
```
