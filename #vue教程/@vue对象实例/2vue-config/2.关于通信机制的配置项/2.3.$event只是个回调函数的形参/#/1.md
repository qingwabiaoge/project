### $event

 vue组件的$event本质是回调函数的 形参 所以可以被赋值任何值 

```vue
    <cl @click="fn($event)"></cl>
----------------------------------
<script>
    var cl = {
        template:`<div @click="click">事件</div>`,
        methods: {
           click:function () {

               this.$emit('click', 123)
           }
        }

    }
    const vm = new Vue({
        el: '#app',
        components: {cl},
        data: {
            msg: 'msg'
        },
        methods:{

            fn($event){
                console.log($event)
            }

        }
    })
    </script>
```
