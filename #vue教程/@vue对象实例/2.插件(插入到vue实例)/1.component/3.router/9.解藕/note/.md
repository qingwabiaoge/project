# 解耦

本质是修改组件的自定义属性的默认值

```js
<meta charset="utf-8">
<script src="https://cdn.bootcss.com/vue/2.5.17-beta.0/vue.js"></script>
<script src="https://cdn.bootcss.com/vue-router/3.0.1/vue-router.min.js"></script>

<div id="app">
  <h1>Route props</h1>
  <ul>
    <li>
      <router-link to="/">/</router-link>
    </li>
    <li>
      <router-link to="/hello/you">/hello/you</router-link>
    </li>
    <li>
      <router-link to="/static">/static</router-link>
    </li>
    <li>
      <router-link to="/dynamic/1">/dynamic/1</router-link>
    </li>

  </ul>

  <hr>

  <router-view class="red" attr="自定义属性"></router-view>
</div>


<script>


  const Hello = {
    template: `<div>
                       <p>
                        name:<b>{{name}}</b> <br>
                        attrs: {{$attrs}}
                        </p>
                    </div>`,
    props: {
      name: {
        type: String,
        default: 'Vue!'
      }
    }
  }


  const router = new VueRouter({
    mode: 'history',

    routes: [
      {path: '/', component: Hello}, // No props, no nothing
      {
        path: '/hello/:name',
        component: Hello,
        props: true
      }, 
      {
        path: '/static',
        component: Hello,
          //修改Hello组件>props>name
        props: {name: 'world'}
      }, 
      //函数生成静态props作为router-view的属性 //修改Hello组件>props>name
      {
        path: '/dynamic/:years',
        component: Hello,
        props(route) {
          return {
            name: (new Date().getFullYear() + parseInt(route.params.years)) + '!'
          }
        }
      }, 
    ]
  })

  new Vue({
    router,


  }).$mount('#app')

</script>

<style>
  .red {
    color: red
  }
</style>

```
